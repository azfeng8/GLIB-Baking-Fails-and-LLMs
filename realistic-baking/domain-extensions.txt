(:action clean-mixer
:parameters (?m - mixer ?t - paper_towel)
:precondition (and
    (wipe-mixer(?m ?t))
    (not (mixer-is-clean ?m))
    (not (towel-is-used ?t))
)
:effect (and
    (mixer-is-clean ?m)
    (towel-is-used ?t)
)
; skill (wipe-mixer ?m ?t)
)

(:action clean-bowl
:parameters (?b - container ?s - soap_hypothetical)
:precondition (and
    (clean ?b ?s)
    (is-bowl ?b)
    (not (soap-consumed ?s))
    (not (container-on-a-stove ?b))
    (not (container-in-an-oven ?b))
)
:effect (and
    (soap-consumed ?s)
    (container-is-clean ?b)
)
; skill (clean ?b ?s)
)
